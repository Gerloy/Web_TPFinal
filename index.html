void Mouse_Menu(){
  if (((mouseX>=PosX1) &&(mouseX<=PosX1+TamX1)) &&
   ((mouseY>=PosY1) && (mouseY<=PosY1+TamY1))){
    pantalla=0;
    iniciado=true;
    if (!cargado){thread("Cargar");}
